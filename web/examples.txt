vector<double> wczytajZPliku() {
    vector<double> liczby;
    ifstream plik(NAZWA_PLIKU);
    if (!plik.is_open()) return liczby;
    string linia;
    while (getline(plik, linia)) {
        if (!linia.empty()) {
            linia.erase(remove_if(linia.begin(), linia.end(), ::isspace), linia.end());
            if (!linia.empty()) {
                try {
                    liczby.push_back(stod(linia));
                } catch (...) {}
            }
        }
    }
    plik.close();
    return liczby;
}

bool zapiszDoPliku(const vector<double>& liczby) {
    ofstream plik(NAZWA_PLIKU);
    if (!plik.is_open()) return false;
    for (size_t i = 0; i < liczby.size(); i++) {
        plik << liczby[i];
        if (i < liczby.size() - 1) plik << endl;
    }
    plik.close();
    return true;
}

void wyswietlZawartosc() {
    vector<double> liczby = wczytajZPliku();
    cout << "\n[0] ZAWARTOSC PLIKU: " << NAZWA_PLIKU << endl;
    cout << "----------------------------------------" << endl;
    if (liczby.empty()) {
        cout << "  PLIK JEST PUSTY" << endl;
        return;
    }
    for (size_t i = 0; i < liczby.size(); i++) {
        cout << "  [" << (i + 1) << "] " << liczby[i] << endl;
    }
    cout << "----------------------------------------" << endl;
    cout << "  LACZNIE: " << liczby.size() << " liczb" << endl;
}

void dodajLiczbe() {
    double liczba;
    cout << "\n[1] DODAJ LICZBE" << endl;
    cout << "----------------------------------------" << endl;
    cout << " Podaj liczbe: ";
    cin >> liczba;
    vector<double> liczby = wczytajZPliku();
    liczby.push_back(liczba);
    if (zapiszDoPliku(liczby)) {
        cout << "  [OK] Dodano liczbe: " << liczba << endl;
    } else {
        cout << "  [BLAD] Nie udalo sie zapisac!" << endl;
    }
}

void usunLiczbe() {
    vector<double> liczby = wczytajZPliku();
    cout << "\n[2] USUN LICZBE" << endl;
    cout << "----------------------------------------" << endl;
    if (liczby.empty()) {
        cout << "  PLIK JEST PUSTY" << endl;
        return;
    }
    wyswietlZawartosc();
    int indeks;
    cout << " Podaj numer do usuniecia (1-" << liczby.size() << "): ";
    cin >> indeks;
    if (indeks < 1 || indeks > (int)liczby.size()) {
        cout << "  [BLAD] Nieprawidlowy numer!" << endl;
        return;
    }
    double usunieta = liczby[indeks - 1];
    liczby.erase(liczby.begin() + indeks - 1);
    if (zapiszDoPliku(liczby)) {
        cout << "  [OK] Usunieto liczbe: " << usunieta << endl;
    }
}

void modyfikujLiczbe() {
    vector<double> liczby = wczytajZPliku();
    cout << "\n[3] ZMODYFIKUJ LICZBE" << endl;
    cout << "----------------------------------------" << endl;
    if (liczby.empty()) {
        cout << "  PLIK JEST PUSTY" << endl;
        return;
    }
    wyswietlZawartosc();
    int indeks;
    cout << " Podaj numer (1-" << liczby.size() << "): ";
    cin >> indeks;
    if (indeks < 1 || indeks > (int)liczby.size()) {
        cout << "  [BLAD] Nieprawidlowy numer!" << endl;
        return;
    }
    double stara = liczby[indeks - 1];
    double nowa;
    cout << " Podaj nowa wartosc: ";
    cin >> nowa;
    liczby[indeks - 1] = nowa;
    if (zapiszDoPliku(liczby)) {
        cout << "  [OK] Zmieniono: " << stara << " -> " << nowa << endl;
    }
}

int partition(vector<double>& arr, int low, int high) {
    double pivot = arr[high];
    int i = low - 1;
    for (int j = low; j < high; j++) {
        if (arr[j] <= pivot) {
            i++;
            swap(arr[i], arr[j]);
        }
    }
    swap(arr[i + 1], arr[high]);
    return i + 1;
}

void quickSort(vector<double>& arr, int low, int high) {
    if (low < high) {
        int pi = partition(arr, low, high);
        quickSort(arr, low, pi - 1);
        quickSort(arr, pi + 1, high);
    }
}

void sortujLiczby() {
    vector<double> liczby = wczytajZPliku();
    cout << "\n[4] SORTUJ (QUICKSORT)" << endl;
    cout << "----------------------------------------" << endl;
    if (liczby.empty()) {
        cout << "  PLIK JEST PUSTY" << endl;
        return;
    }
    quickSort(liczby, 0, (int)liczby.size() - 1);
    if (zapiszDoPliku(liczby)) {
        cout << "  [OK] Posortowano rosnaco!" << endl;
        cout << "  Pierwsze 5: ";
        for (size_t i = 0; i < min((size_t)5, liczby.size()); i++) {
            cout << liczby[i] << " ";
        }
        cout << endl;
    }
}

void obliczSrednia() {
    vector<double> liczby = wczytajZPliku();
    cout << "\n[5] OBLICZ SREDNIA" << endl;
    cout << "----------------------------------------" << endl;
    if (liczby.empty()) {
        cout << "  PLIK JEST PUSTY" << endl;
        return;
    }
    double suma = 0;
    for (double l : liczby) suma += l;
    double srednia = suma / liczby.size();
    cout << "  Ilosc: " << liczby.size() << endl;
    cout << "  Suma: " << suma << endl;
    cout << "  SREDNIA: " << srednia << endl;
}

void wyszukajLiczbe() {
    vector<double> liczby = wczytajZPliku();
    cout << "\n[6] WYSZUKAJ LICZBE" << endl;
    cout << "----------------------------------------" << endl;
    if (liczby.empty()) {
        cout << "  PLIK JEST PUSTY" << endl;
        return;
    }
    double szukana;
    cout << " Podaj szukana liczbe: ";
    cin >> szukana;
    vector<int> pozycje;
    for (size_t i = 0; i < liczby.size(); i++) {
        if (abs(liczby[i] - szukana) < 0.0001) {
            pozycje.push_back(i + 1);
        }
    }
    if (pozycje.empty()) {
        cout << "  [NIE ZNALEZIONO]" << endl;
    } else {
        cout << "  [ZNALEZIONO] Na pozycjach: ";
        for (int p : pozycje) cout << p << " ";
        cout << endl;
    }
}

void zliczWystapienia() {
    vector<double> liczby = wczytajZPliku();
    cout << "\n[7] ZLICZ WYSTAPIENIA" << endl;
    cout << "----------------------------------------" << endl;
    if (liczby.empty()) {
        cout << "  PLIK JEST PUSTY" << endl;
        return;
    }
    double szukana;
    cout << " Podaj liczbe: ";
    cin >> szukana;
    int ile = 0;
    for (double l : liczby) {
        if (abs(l - szukana) < 0.0001) ile++;
    }
    cout << "  Wystapienia '" << szukana << "': " << ile << " razy" << endl;
}

void usunDuplikaty() {
    vector<double> liczby = wczytajZPliku();
    cout << "\n[8] USUN DUPLIKATY" << endl;
    cout << "----------------------------------------" << endl;
    if (liczby.empty()) return;
    int przed = liczby.size();
    vector<double> bez;
    vector<bool> seen(liczby.size(), false);
    for (size_t i = 0; i < liczby.size(); i++) {
        if (!seen[i]) {
            bez.push_back(liczby[i]);
            for (size_t j = i + 1; j < liczby.size(); j++) {
                if (abs(liczby[j] - liczby[i]) < 0.0001) seen[j] = true;
            }
        }
    }
    if (zapiszDoPliku(bez)) {
        cout << "  [OK] Usunieto " << (przed - bez.size()) << " duplikatow" << endl;
    }
}

void minMax() {
    vector<double> liczby = wczytajZPliku();
    cout << "\n[9] MINIMUM / MAKSIMUM" << endl;
    cout << "----------------------------------------" << endl;
    if (liczby.empty()) {
        cout << "  PLIK JEST PUSTY" << endl;
        return;
    }
    double min_val = liczby[0], max_val = liczby[0];
    for (double l : liczby) {
        if (l < min_val) min_val = l;
        if (l > max_val) max_val = l;
    }
    cout << "  MINIMUM: " << min_val << endl;
    cout << "  MAKSIMUM: " << max_val << endl;
}

void kopiujPlik() {

    cout << " Podaj nazwe kopii: ";
    cin >> nowaNazwa;
    ifstream zrodlo(NAZWA_PLIKU);
    ofstream cel(nowaNazwa);

        cout << "  [BLAD] Nie mozna utworzyc kopii!" << endl;
        return;
    }
    cel << zrodlo.rdbuf();
    cout << "  [OK] Utworzono kopie: " << nowaNazwa << endl;
}
