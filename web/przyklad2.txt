#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

struct Student {
    string imie;
    string nazwisko;
    int numer;
    double srednia;
};

void wyswietlMenu() {
    cout << "==============================" << endl;
    cout << "   ZARZADZANIE STUDENTAMI" << endl;
    cout << "==============================" << endl;
    cout << "1. Dodaj studenta" << endl;
    cout << "2. Wyswietl studentow" << endl;
    cout << "3. Szukaj studenta" << endl;
    cout << "4. Usun studenta" << endl;
    cout << "5. Sortuj wedlug numeru" << endl;
    cout << "6. Sortuj wedlug sredniej" << endl;
    cout << "7. Statystyki" << endl;
    cout << "0. Wyjscie" << endl;
    cout << "==============================" << endl;
    cout << "Wybierz opcje: ";
}

void dodajStudenta(vector<Student>& studenci) {
    Student s;
    cout << "Podaj imie: ";
    cin >> s.imie;
    cout << "Podaj nazwisko: ";
    cin >> s.nazwisko;
    cout << "Podaj numer indeksu: ";
    cin >> s.numer;
    cout << "Podaj srednia ocen: ";
    cin >> s.srednia;
    
    studenci.push_back(s);
    cout << "Dodano studenta: " << s.imie << " " << s.nazwisko << endl;
}

void wyswietlStudentow(const vector<Student>& studenci) {
    if (studenci.empty()) {
        cout << "Brak studentow w bazie." << endl;
        return;
    }
    
    cout << "\n===== LISTA STUDENTOW =====" << endl;
    cout.width(4);  cout << "LP.";
    cout.width(12); cout << "Imie";
    cout.width(12); cout << "Nazwisko";
    cout.width(10); cout << "Numer";
    cout.width(10); cout << "Srednia" << endl;
    cout << "-----------------------------------" << endl;
    
    for (size_t i = 0; i < studenci.size(); i++) {
        cout.width(4);  cout << i + 1 << ".";
        cout.width(12); cout << studenci[i].imie;
        cout.width(12); cout << studenci[i].nazwisko;
        cout.width(10); cout << studenci[i].numer;
        cout.width(10); cout << studenci[i].srednia << endl;
    }
    cout << "-----------------------------------" << endl;
    cout << "Lacznie: " << studenci.size() << " studentow" << endl;
}

void szukajStudenta(const vector<Student>& studenci) {
    if (studenci.empty()) {
        cout << "Brak studentow w bazie." << endl;
        return;
    }
    
    string nazw;
    cout << "Podaj nazwisko do wyszukania: ";
    cin >> nazw;
    
    bool znaleziono = false;
    cout << "\n===== WYNIKI WYSZUKIWANIA =====" << endl;
    
    for (const auto& s : studenci) {
        if (s.nazwisko == nazw) {
            cout << s.imie << " " << s.nazwisko;
            cout << " | Numer: " << s.numer;
            cout << " | Srednia: " << s.srednia << endl;
            znaleziono = true;
        }
    }
    
    if (!znaleziono) {
        cout << "Nie znaleziono studenta o nazwisku: " << nazw << endl;
    }
}

void usunStudenta(vector<Student>& studenci) {
    if (studenci.empty()) {
        cout << "Brak studentow do usuniecia." << endl;
        return;
    }
    
    int numer;
    cout << "Podaj numer indeksu do usuniecia: ";
    cin >> numer;
    
    for (auto it = studenci.begin(); it != studenci.end(); ++it) {
        if (it->numer == numer) {
            cout << "Usunieto studenta: " << it->imie << " " << it->nazwisko << endl;
            studenci.erase(it);
            return;
        }
    }
    cout << "Nie znaleziono studenta o numerze: " << numer << endl;
}

bool porownajNumer(const Student& a, const Student& b) {
    return a.numer < b.numer;
}

bool porownajSrednia(const Student& a, const Student& b) {
    return a.srednia > b.srednia;
}

void statystyki(const vector<Student>& studenci) {
    if (studenci.empty()) {
        cout << "Brak studentow." << endl;
        return;
    }
    
    double suma = 0;
    double min_srednia = studenci[0].srednia;
    double max_srednia = studenci[0].srednia;
    string najlepszy = studenci[0].nazwisko;
    string najgorszy = studenci[0].nazwisko;
    
    for (const auto& s : studenci) {
        suma += s.srednia;
        if (s.srednia > max_srednia) {
            max_srednia = s.srednia;
            najlepszy = s.nazwisko;
        }
        if (s.srednia < min_srednia) {
            min_srednia = s.srednia;
            najgorszy = s.nazwisko;
        }
    }
    
    double srednia_wszystkich = suma / studenci.size();
    
    cout << "\n===== STATYSTYKI =====" << endl;
    cout << "Liczba studentow: " << studenci.size() << endl;
    cout << "Srednia wszystkich: " << srednia_wszystkich << endl;
    cout << "Najwyzsza srednia: " << max_srednia << " (" << najlepszy << ")" << endl;
    cout << "Najnizsza srednia: " << min_srednia << " (" << najgorszy << ")" << endl;
}

int main() {
    vector<Student> studenci;
    int wybor;
    
    cout << "\n================================" << endl;
    cout << "   SYSTEM ZARZADZANIA STUDENTAMI" << endl;
    cout << "================================" << endl;
    
    do {
        wyswietlMenu();
        cin >> wybor;
        
        switch (wybor) {
            case 1:
                dodajStudenta(studenci);
                break;
            case 2:
                wyswietlStudentow(studenci);
                break;
            case 3:
                szukajStudenta(studenci);
                break;
            case 4:
                usunStudenta(studenci);
                break;
            case 5:
                sort(studenci.begin(), studenci.end(), porownajNumer);
                cout << "Posortowano wedlug numeru indeksu." << endl;
                break;
            case 6:
                sort(studenci.begin(), studenci.end(), porownajSrednia);
                cout << "Posortowano wedlug sredniej (od najwyzszej)." << endl;
                break;
            case 7:
                statystyki(studenci);
                break;
            case 0:
                cout << "Do widzenia!" << endl;
                break;
            default:
                cout << "Nieznana opcja!" << endl;
        }
        cout << endl;
        
    } while (wybor != 0);
    
    return 0;
}
